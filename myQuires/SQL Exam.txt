1. SELECT FIRST_NAME AS NAME,LAST_NAME AS SURNAME
   FROM WORKER_Details
   WHERE SALARY IN (SELECT SALARY 
                FROM WORKER_Details
				GROUP BY SALARY
				HAVING COUNT(*)>1);


2.SELECT FIRST_NAME 
  FROM WORKER_DETAILS
  WHERE SALARY IN (SELECT  MAX(SALARY) AS TOTAL
                    FROM WORKER_DETAILS
                    GROUP BY DEPARTMENT);
